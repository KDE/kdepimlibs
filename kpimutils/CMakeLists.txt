project(kpimutils)

add_definitions( -DQT_NO_CAST_FROM_ASCII )

  ../kmime
  ${CMAKE_CURRENT_BINARY_DIR}/../kmime
)

  emailvalidator.cpp
  linklocator.cpp
  spellingfilter.cpp
